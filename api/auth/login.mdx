---
title: Log in
description: Authenticate and log in to an existing account
---

## Endpoint

```
POST /api/auth/login
```

**Authentication:** Not required

## Request body

<ParamField name="email" type="string" required>
  Email address of the account
</ParamField>

<ParamField name="password" type="string" required>
  Password for the account
</ParamField>

## Response

<ResponseField name="success" type="boolean">
  Indicates if the request was successful
</ResponseField>

<ResponseField name="userData" type="object">
  The authenticated user object
  
  <ResponseField name="_id" type="string">
    Unique user identifier
  </ResponseField>
  
  <ResponseField name="fullName" type="string">
    Full name of the user
  </ResponseField>
  
  <ResponseField name="email" type="string">
    Email address of the user
  </ResponseField>
  
  <ResponseField name="bio" type="string">
    User biography
  </ResponseField>
  
  <ResponseField name="profilePic" type="string">
    URL to profile picture
  </ResponseField>
  
  <ResponseField name="createdAt" type="string">
    Account creation timestamp
  </ResponseField>
  
  <ResponseField name="updatedAt" type="string">
    Last update timestamp
  </ResponseField>
</ResponseField>

<ResponseField name="token" type="string">
  JWT authentication token
</ResponseField>

<ResponseField name="message" type="string">
  Success or error message
</ResponseField>

## Example request

```bash
curl -X POST https://api.quickchat.com/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{
    "email": "john@example.com",
    "password": "securepassword123"
  }'
```

## Error responses

### User not found

```json
{
  "success": false,
  "message": "User not found"
}
```

### Invalid credentials

```json
{
  "success": false,
  "message": "Invalid credentials"
}
```
